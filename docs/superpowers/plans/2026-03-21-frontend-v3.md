```tsx
import { type ClassValue, clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

export function formatDate(dateStr: string): string {
  const date = new Date(dateStr);
  const now = new Date();
  const diffMs = now.getTime() - date.getTime();
  const diffMin = Math.floor(diffMs / 60000);
  if (diffMin < 1) return 'just now';
  if (diffMin < 60) return `${diffMin}m ago`;
  const diffHrs = Math.floor(diffMin / 60);
  if (diffHrs < 24) return `${diffHrs}h ago`;
  const diffDays = Math.floor(diffHrs / 24);
  if (diffDays < 7) return `${diffDays}d ago`;
  return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
}

export function formatDuration(ms: number): string {
  if (ms < 1000) return `${ms}ms`;
  const seconds = ms / 1000;
  if (seconds < 60) return `${seconds.toFixed(1)}s`;
  const minutes = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${minutes}m ${secs}s`;
}

export function truncate(str: string, maxLen: number): string {
  if (str.length <= maxLen) return str;
  return str.slice(0, maxLen - 1) + '\u2026';
}

export function getStatusColor(status: string): string {
  switch (status) {
    case 'done': return 'success';
    case 'failed': return 'error';
    case 'queued': return 'secondary';
    default: return 'warning'; // all active statuses
  }
}
```

```tsx
import { API_BASE_URL } from './constants';
import type { RunSummary, RunResult, RunLaunch, ToolTrace, RunConstraints } from '@/types';

export class ApiError extends Error {
  constructor(public status: number, message: string) {
    super(message);
    this.name = 'ApiError';
  }
}

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${API_BASE_URL}${path}`, {
    ...init,
    headers: { 'Content-Type': 'application/json', ...init?.headers },
    cache: 'no-store',
  });
  if (!res.ok) {
    const detail = await res.text().catch(() => `Request failed: ${res.status}`);
    throw new ApiError(res.status, detail);
  }
  return res.json() as Promise<T>;
}

export const api = {
  healthz: () => request<{ status: string }>('/healthz'),

  listRuns: () => request<RunSummary[]>('/v1/runs'),

  getRun: (id: string) => request<RunResult>(`/v1/runs/${id}`),

  getTrace: (id: string) => request<ToolTrace[]>(`/v1/runs/${id}/trace`),

  getReport: async (id: string): Promise<string> => {
    const res = await fetch(`${API_BASE_URL}/v1/runs/${id}/report.md`, { cache: 'no-store' });
    if (!res.ok) throw new ApiError(res.status, await res.text().catch(() => 'Failed'));
    return res.text();
  },

  launchRun: (payload: { topic: string; constraints?: RunConstraints }) =>
    request<RunLaunch>('/v1/runs/launch', { method: 'POST', body: JSON.stringify(payload) }),

  createRun: (payload: { topic: string; constraints?: RunConstraints }) =>
    request<RunResult>('/v1/runs', { method: 'POST', body: JSON.stringify(payload) }),

  rerunPlanner: (id: string) =>
    request<RunResult>(`/v1/runs/${id}/planner/rerun`, { method: 'POST' }),
} as const;
```

```tsx
export const topicSchema = z.object({

export const constraintsSchema = z.object({

  open_access_only: z.boolean(),
  max_selected_papers: z.number().int().min(1).max(100),

  lab_mode: z.enum(['wet', 'dry', 'either']),

  { message: 'Start year must be before end year', path: ['year_from'] },
```

```tsx
import { useQuery } from '@tanstack/react-query';
import { useRef, useEffect } from 'react';
import { toast } from 'sonner';
import { api } from '@/lib/api-client';
import { POLLING_INTERVAL_MS } from '@/lib/constants';

export function usePollRun(runId: string) {
  const failCount = useRef(0);
  const toastId = useRef<string | number | undefined>();

  const query = useQuery({
    queryKey: ['run', runId],
    queryFn: async () => {
      try {
        const result = await api.getRun(runId);
        if (failCount.current > 0) {
          failCount.current = 0;
          if (toastId.current) toast.dismiss(toastId.current);
        }
        return result;
      } catch (err) {
        failCount.current += 1;
        if (failCount.current >= 3 && !toastId.current) {
          toastId.current = toast.error('Connection lost', {
            description: 'Having trouble reaching the server. Retrying...',
            duration: Infinity,
          });
        }
        throw err;
      }
    },
    enabled: !!runId,
    refetchInterval: (query) => {
      const status = query.state.data?.status;
      if (status === 'done' || status === 'failed') return false;
      return POLLING_INTERVAL_MS;
    },
    retry: false, // we handle retries via refetchInterval
  });

  // Cleanup toast on unmount
  useEffect(() => {
    return () => { if (toastId.current) toast.dismiss(toastId.current); };
  }, []);

  return query;
}
```

```tsx
import { useEffect, useState } from 'react';

export function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(false);
  useEffect(() => {
    const mql = window.matchMedia(query);
    setMatches(mql.matches);
    const handler = (e: MediaQueryListEvent) => setMatches(e.matches);
    mql.addEventListener('change', handler);
    return () => mql.removeEventListener('change', handler);
  }, [query]);
  return matches;
}
```

```tsx
import { create } from 'zustand';

type ItemType = 'hypothesis' | 'conflict' | 'evidence' | 'paper';

interface DossierState {
  selectedType: ItemType | null;
  selectedId: string | null;
  searchQuery: string;
  expandedGroups: Record<string, boolean>;
  select: (type: ItemType, id: string) => void;
  clearSelection: () => void;
  setSearchQuery: (q: string) => void;
  toggleGroup: (group: string) => void;
  expandGroup: (group: string) => void;
  reset: () => void;
}

const initialState = {
  selectedType: null as ItemType | null,
  selectedId: null as string | null,
  searchQuery: '',
  expandedGroups: { hypotheses: true, conflicts: true, evidence: true, papers: false } as Record<string, boolean>,
};

export const useDossierStore = create<DossierState>((set) => ({
  ...initialState,
  // Convention: for hypotheses, selectedId is String(rank), e.g. "1", "2", "3"
  // For conflicts: cluster_id. For evidence: evidence_id. For papers: paper_id.
  select: (type, id) => set({ selectedType: type, selectedId: id }),
  clearSelection: () => set({ selectedType: null, selectedId: null }),
  setSearchQuery: (q) => set({ searchQuery: q }),
  toggleGroup: (group) =>
    set((state) => ({
      expandedGroups: { ...state.expandedGroups, [group]: !state.expandedGroups[group] },
    })),
  expandGroup: (group) =>
    set((state) => ({
      expandedGroups: { ...state.expandedGroups, [group]: true },
    })),
  reset: () => set(initialState),
}));
```
